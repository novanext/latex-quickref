\documentclass[12pt,a5paper]{article}

\usepackage[english]{babel} % proper hyphenation for english
\usepackage[utf8]{inputenc} % allows typing of non-unicode characters
\usepackage[T1]{fontenc} % for better display of non ascii characters in output pdf
\usepackage[small,bf,hang]{caption} % nicer captions for figures and tables
\usepackage{fourier} % sets the font to Utopia and provides additional mathematical symbols
\usepackage{booktabs} % feature for elegant tables with \toprule, \midrule and \bottomrule
\usepackage{graphicx} % for including figures
\usepackage{amsmath} % extended mathematics functionality (\eqref{,\text, etc})
\usepackage{physics}
\usepackage[separate-uncertainty=true]{siunitx} % correct typesetting of numbers and units
\usepackage{todonotes} % allows you to insert 'todo' annotations
\usepackage{subcaption} % sllows you to insert sub-figures and captions
\usepackage{xcolor} % for the use of colors
\definecolor{darkblue}{rgb}{0,0,0.5} % defines color 'darkblue'
\usepackage[numbers]{natbib} % advanced bibliographic tools
% hyperref: makes all citations, references and urls clickable
\usepackage[colorlinks=true, urlcolor=darkblue, anchorcolor=darkblue, filecolor=green, linkcolor=darkblue, menucolor=darkblue,citecolor=darkblue,pdfpagemode=UseOutlines, pdfstartview=FitH,hypertexnames={false},bookmarksopen=true,breaklinks=true,pdfencoding=unicode]{hyperref}
\usepackage{doi} % makes the DOI in references into hyperlinks
\usepackage{cleveref} %gives you the \cref -like commands for nicer cross referencing

\begin{document}
The packages have been successfully installed.

\end{document}
